   export function useWeakAreas() {
     return useQuery({
       queryKey: ['weakAreas'],
       queryFn: async () => {
         const { data, error } = await client.GET('/api/v1/practice/weak-areas');
         if (error) throw error;
         return data;
       },
     });
   }

   ```typescript
   export function useCreatePractice() {
     return useMutation({
       mutationFn: async ({ weak_topics, drill_count = 5 }: { weak_topics: string[]; drill_count?: number }) => {
         const { data, error } = await client.POST('/api/v1/practice', {
           body: { weak_topics, drill_count },
         });
         if (error) throw error;
         return data;
       },
     });
   }
   ```
