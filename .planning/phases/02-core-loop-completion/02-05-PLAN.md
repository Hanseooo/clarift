   ```typescript
   export function useCreateQuiz() {
     return useMutation({
       mutationFn: async ({ document_id, question_count = 5, auto_mode = true }: { document_id: string; question_count?: number; auto_mode?: boolean }) => {
         const { data, error } = await client.POST('/api/v1/quizzes', {
           body: { document_id, question_count, auto_mode },
         });
         if (error) throw error;
         return data;
       },
     });
   }
   ```
5. Define `useSubmitAttempt`:
   ```typescript
   export function useSubmitAttempt() {
     return useMutation({
       mutationFn: async ({ quiz_id, answers }: { quiz_id: string; answers: Record<string, string> }) => {
         const { data, error } = await client.POST('/api/v1/quizzes/attempts', {
           body: { quiz_id, answers },
         });
         if (error) throw error;
         return data;
       },
     });
   }
   ```
