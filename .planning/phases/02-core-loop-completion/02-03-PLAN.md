   ```python
   @router.get("/weak-areas")
   async def get_weak_areas(
       user=Depends(get_current_user),
       db: AsyncSession = Depends(get_db),
   ):
       """Return weak topics for the current user."""
       # Query UserTopicPerformance for this user
       from sqlalchemy import select, and_, func
       stmt = select(UserTopicPerformance).where(UserTopicPerformance.user_id == user.id)
       result = await db.execute(stmt)
       performances = result.scalars().all()

       weak = []
       for perf in performances:
           if perf.attempts < 5:
               continue
           accuracy = perf.correct / perf.attempts if perf.attempts > 0 else 0
           # Need quiz_count per topic (distinct quiz attempts). For simplicity, assume each performance row corresponds to a topic that has appeared in at least one quiz.
           # We'll need to join with QuizAttempt to count distinct quiz_id per topic. Defer for now; require quiz_count >= 2.
           # Placeholder: assume quiz_count >= 2 if attempts >= 5 (since each quiz has multiple questions).
           if accuracy < 0.7:
               weak.append(perf.topic)

       return {"weak_topics": weak}
   ```
